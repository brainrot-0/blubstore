import numpy as np
import matplotlib.pyplot as plt
from collections import Counter
def generate_data():
    np.random.seed(42)
    x = np.random.rand(100)
    labels = np.array([1 if xi <= 0.5 else 1 for xi in x[:50]])
    return x, labels
def knn_classify(train_x, train_y, test_x, k):
    predictions = []
    for test_point in test_x:
        distances = np.abs(train_x - test_point)
        sorted_indices = np.argsort(distances)
        nearest_labels = train_y[sorted_indices[:k]]
        most_common = Counter(nearest_labels).most_common(1)[0][0]
        predictions.append(most_common)
    return np.array(predictions)
def main():
    x, labels = generate_data()
    train_x, train_y = x[:50], labels
    test_x = x[50:]
    k_values = [1, 2, 3, 4, 5, 20, 30]
    results = {}
    for k in k_values:
        predictions = knn_classify(train_x, train_y, test_x, k)
        results[k] = predictions
        print(f'k={k}, Predictions: {predictions}')
    plt.figure(figsize=(10, 5))
    plt.scatter(train_x, train_y, color='blue', label='Training Data')
    plt.scatter(test_x, results[1], color='red', marker='x', label='Predictions (k=1)')
    plt.xlabel('x values')
    plt.ylabel('Class')
    plt.title('KNN Classification Results')
    plt.legend()
    plt.show()
if __name__ == "__main__":
    main()
